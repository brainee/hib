基本参数说明


    @param template     {String}    模板视图文本
    @param needMask     {Bool}      是否需要mask
    @param template     {String}    模板视图文本
    @param events       {Object}    绑定事件
    @param setDatamodel {Object}    动态设置属性、事件
    @param datamodel    {
      data:      {Array}            数据
      upClass:   {String}           向上的class
      upClass:   {String}           向上的class
      upClass:   {String}           向上的class
      index:     {Int}              选择的下标
      dir:       {String}           箭头的方向
    }

demo1


if(!this.demo1) {
    var data = [{ name: '<span class="center">价格/星级</span>' },
      { name: '<span class="center">位置区域</span>' },
      { name: '<span class="center">品牌</span>' },
      { name: '<span class="center">测试</span>'}];


    var el = $(e.currentTarget);
    var index = parseInt(Math.random() * 4);
    var dir = (e.pageY > 200) ? 'down' : 'up';
    this.demo1 = new UIBubbleLayer({
      triggerEl: el,
      datamodel: {
        data: data,
        dir: dir,
        index: index
      },
      onClick:  function (e, data, index, el) {
        console.log(arguments);
        this.setIndex(index);
        this.hide();
      }
    });
  }
  this.demo1.show();
