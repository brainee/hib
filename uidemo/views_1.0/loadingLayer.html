这里是c.ui.loadingLayer的使用，该插件作为数据加载的过渡层,可以配置文本

主要属性如下： 

// @param contentDom {dom} underscore提供的模板方法
// @param callback {function} 回调函数
// @param message {String} 内容体

demo：loadingLayer的简单初始化

define(['app/ui/c.ui.loadingLayer'], function(LoadingLayer){
            var l = new LoadingLayer(function () {
                    alert('执行的内容');//点击X按钮后的回调函数
                });
            l.show();// @description 显示loadingLayer框
});
