这里是c.ui.alert的使用，该插件提供提供类似于alert的弹出层 

// @param tpl {function}                underscore提供的模板方法
    // @param title {String}                标题
    // @param message {String}              内容体
    // @param buttons {array}               需要生成的按钮
    // @param viewdata {object}             模板视图文本

1、简便写法：

javascript: this.showMessage('简便写法');

2、初始化写法：

define(['app/ui/c.ui.alert'], function(Alert){
    this.alert = new cUI.Alert({
    title: '提示信息',
    message: '测试信息',//控件信息内容，可以自定义
    buttons: [{
    text: '取消',
    click: function () {
    this.hide();
    }
    }
    ]
    });
    alert.show()
    });

3、设置多个按钮：

define(['app/ui/c.ui.alert'], function(Alert){
    this.alert = new cUI.Alert({
    title: '提示信息',
    message: '测试信息',
    buttons: [{
    text: '取消',
    click: function () {
    this.hide();
    }
    }，{
    text: '确定',
    click: function () {
    this.hide();
    }
    }
    ]
    });
    alert.show()
    });
