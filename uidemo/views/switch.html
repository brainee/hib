初始化：

define(['app/ui/c.ui.switch'], function(_Switch){
    var switch = new _Switch({
        rootBox: this.$el.find('.hm'),    //switch的容器
        checked: true,                    //默认状态
        changed: function () {          //状态发生改变时执行的函数
            $('#msg').html($('< div>当前状态：' + this.getStatus() + '< /div>'));
        }
    });
});

注意：switch的容器，建议该容器设为display:inline-block;
