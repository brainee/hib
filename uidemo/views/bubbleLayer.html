主要属性如下：

// @param dir{String}         el相对于气泡显示方向：up/down
// @param triggerEl{object}         触发事件的el对象
// @param click{function}         el点击事件
// @param itemTemplate{String}         显示气泡的html模板(underscore模板)
// @param data{object}         传入模板的参数对象

Demo：

define(['libs', 'c', 'cBasePageView'], function(libs, c, BasePageView){
    this.showBubbleLayer({
        dir: 'up',
        triggerEl: $(e.currentTarget),
        click: function(data){
            self.showMessage(JSON.stringify(data));
        },
        itemTemplate: '<%=itemData.name%>：<%=itemData.data%>',
        data: [
            { name: '价格', data: '￥35' },
            { name: '评分', data: 80 },
            { name: '级别', data: 5 }
        ]
    });
});
