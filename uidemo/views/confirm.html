confirm初始化：

define(['app/ui/c.ui.alert'], function(Alert){
    this.alert = new cUI.Alert({
        title: 'confirm title',
        message: 'confirm test',//控件信息内容，可以自定义
        type: 'confirm'
    });
    alert.show()
});
