这里是c.ui.loading的使用，该插件作为数据加载的过渡层

主要属性如下：

// @param html {String}         自定义class

1、loading的简单初始化：

define(['cUILoading'], function(Loading){

    var loading = new Loading();
    // --------------------
    // 显示Loading框
    loading.show();

    // --------------------
    // 业务逻辑 -- 加载数据
    loading.hide();
});

2、带关闭按钮loading初始化：

define(['cUILoading'], function(LoadingLayer){
    var loadingLayer = new c.ui.LoadingLayer(function () {
        alert('执行的内容');
        var s = '';
    },"自定义文字内容");
    loadingLayer.show();
});
