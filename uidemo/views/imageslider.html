主要属性如下：

// @param container {dom}            容器元素的dom
// @param images {Array}             轮播图片数组 array [object] object :{src,alt,onClick,}
// @param autoPlay {bool}            是否自动播放
// @param index {int}                当前图片索引
// @param delay {int}                自动播放延迟(单位：毫秒)
// @param dir {String}              轮播方向
// @param onChanged {function}       轮播开始的事件，返回数据显示的图片信息，当前显示的索引(从0开始)
// @param onImageClick {function}    图片点击事件
// @param scope {object}             点击事件上下文
// @param imageSize{object}          原autoHeight属性不公开,如果设置必须，宽和高同时设置，如果不设置则根据图片大小等比缩放[object] object :{width,height}    
// @param showNav boolean            是否显示导航
// @param errorImageUrl {string}     图片加载错误显示的图片
// @param defaultImageUrl {string}   没有图片数据时显示的图片

插件引用

    define(['cUIImageSlider']);

几种方法

//开始播放
imageSlider.play();
//停止播放
imageSlider.stop();
//重新播放
imageSlider.rePlay();
//跳转至下一张
imageSlider.next();
//跳转至上一张
imageSlider.pre();

实例化对象

this.imageSlider = new imageSlider({
    //图片数组
    images: [ {
        //图片路径，绝对地址
        src: "http://res.m.ctrip.com/market/appimg/h5/20140305/3.jpg",
        //点击图片的事件
        onClick: $.proxy(function (imageInfo) {
            this._onSelectedImageClick(imageInfo);
        },this),
        //onClick事件的上下文
        scope: this },
            { src: "http://res.m.ctrip.com/market/appimg/h5/20140305/2.jpg"},
            { src: "http://res.m.ctrip.com/market/appimg/h5/20140305/1.jpg"}
    ],
    //容器
    container:$("#imageSliderDemo"),
    //滚动方向,默认LEFT
    dir: "LEFT",
    //默认显示的图片索引
    index: 0,
    //是否自动播放
    noNeedDefault:'false'，
    //是否要阻止浏览器默认行为
    autoPlay: true,
    //所有图片的单击事件，如果对应的图片设置了自己的click事件，该事件则不执行
    onImageClick: $.proxy(function
    (imageInfo) { this._onImageClick(); }, this)
});
this.imageSlider.paly();
