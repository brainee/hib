引用方式

   
      define([
        'cUIScroll'
      ], function(Scroll) {
      });

调用方式

  
        new Scroll({
           wrapper: this.$('#wrapper')
        });

计算Scroller宽度

  
        var wrapper = this.$('#wrapper');  //获取wrapper
        var scroller = wrapper.children().eq(0);  //获取wrapper的第一个元素
        var items = scroller.find('li');  //获取scroller的li子元素
        var item = items.eq(0);  //获取第一个li元素
        var itemWidth = item.width() + 10;  //计算每个li的宽度，10可理解为margin值
        var scrollerWidth = items.length * itemWidth;  //scroller的宽度

        scroller.css('width', scrollerWidth);

默认参数

  
      {
           wrapper: null,   //外层的包裹器， 必须有
           scroller：null,  //内层的内容， 默认为wrapper的第一个子元素,可以不传
           //是否具有滚动条
           scrollbars: true,
           //初始坐标
           startX: 0,
           startY: 0,
           scrollX: false, //横向滚动
           scrollY: true,   //竖直滚动
           scrollStart: function() {}, //滚动开始
           scrollEnd： function() {} //滚动结束
        }

api

  
if (!this.uiScroll) {
  this.uiScroll = new Scroll({})
  this.uiScroll.refresh(); //当滑动框的高度或者宽度变化的时候，调用这个方法
  this.uiScroll.x //获取当前的x坐标
  this.uiScroll.maxScrollX // 获取做大的横坐标
  this.uiScroll.distX   // 与0比较，判断是向左滑动，还是向右
  this.uiScroll.scrollTo(x, y, time); //滑到指定位置,这个api不会触发scrollEnd和scrollStart
  //当需要支持响应式设计的时候，windows resize, 注意事件不要重复添加
  this.uiScroll.on('refresh', function() {
    console.log('refresh');
  });
}
